public class FigModifyingModeImpl extends ModeImpl implements FigModifyingMode {
    protected Editor editor;
}

public class ModeSelect extends FigModifyingModeImpl {
    private Point selectAnchor;

    public void mousePressed(MouseEvent me) {
        Fig underMouse = editor.hit(selectAnchor);
    }
}

public class Editor implements Serializable, MouseListener,
        MouseMotionListener, KeyListener {
    protected static Rectangle _hitRect;

    public final Fig hit(Point p) {
        return hit(_hitRect);
    } //return to the caller

    public Fig hit(Rectangle r) {
        Fig f = getLayerManager().hit(r);
        return f;
    } //return to the caller
}

public class LayerManager implements java.io.Serializable {
    private Layer _activeLayer;

    public Fig hit(Rectangle r) {
        return _activeLayer.hit(r);
    } //return to the caller
}

public class LayerDiagram extends Layer {
    private List<Fig> contents;

    public Fig hit(Rectangle r) {
        int i;
        Fig f = (Fig) contents.get(i);
        return f;
    } //return to the caller
}
